-FirstName
-LastName
-PIN
-AccessLevel
-AuthorizedUntilDate
-Position
-Phone
-Email

CREATE TABLE [dbo].[Employee]
(
	[InternalEmpId] INT NOT NULL IDENTITY(1,1),
	[EmployeeId] AS 'EM' + RIGHT('000' + CAST( [InternalEmpId] AS VARCHAR(10)),3) PERSISTED NOT NULL,
	[FirstName] NVARCHAR(50) NOT NULL,
	[LastName] NVARCHAR(50) NOT NULL,
	[PIN] INT NOT NULL,
	[AccessLevel] INT NOT NULL,
	[AuthorizedUntilDate] DateTime NOT NULL,
	[Position] NVARCHAR(50) NOT NULL,
	[Phone] NVARCHAR(50) NOT NULL,
	[Email] NVARCHAR(50) NOT NULL,
	
	CONSTRAINT PK_Employee PRIMARY KEY CLUSTERED(EmployeeId)

)

History
-HistoryId
-DoorId {FK}
-EmployeeId {FK}
-DateTimeStamp

CREATE TABLE [dbo].[AccessHistory] (
    [InternalHistoryId]     INT      IDENTITY (1, 1) NOT NULL,
	[HistoryId] AS 'HS' + RIGHT('000' + CAST( [InternalHistoryId] AS VARCHAR(10)),3) PERSISTED NOT NULL,
    [DoorId]        NVARCHAR(50)      NOT NULL,
    [EmployeeId]    NVARCHAR(50)      NULL,
    [VisitorId]     NVARCHAR(50)      NULL,
    [DateTimeStamp] DATETIME NOT NULL,
    CONSTRAINT PK_AccessHistoruy PRIMARY KEY CLUSTERED(HistoryId),
    CONSTRAINT [FK_DoorId] FOREIGN KEY ([DoorId]) REFERENCES [dbo].[DoorTerminal] ([DoorId]),
    CONSTRAINT [FK_VisitorId] FOREIGN KEY ([VisitorId]) REFERENCES [dbo].[Visitor] ([VisitorId]),
    CONSTRAINT [FK_EmployeeId] FOREIGN KEY ([EmployeeId]) REFERENCES [dbo].[Employee] ([EmployeeId])
)
